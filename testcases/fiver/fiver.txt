define comp square(x, y) isa box()
{
	var size = 20
	var start = 40

	id(("square"+x)+y)
	position-x(x * size + start)
	position-y(y * size + start)
	width(size)
	height(size)
	border(1)
	border-color("#000000")
	background-color("#FFFFFF")
}
